# Grfico 1: Evolucin en el tiempo de un solo indicador para un solo pas o regin
def plot_single_indicator(data, indicator_name, country_or_region):
    selected_data = data[(data['Series Name'] == indicator_name) & (data['Country Name'] == country_or_region)]
    
    if len(selected_data.columns) > 4 and len(selected_data.iloc[0, 4:].dropna()) > 0:
        plt.figure(figsize=(10, 6))
        plt.plot(selected_data.columns[4:], selected_data.iloc[0, 4:])
        plt.title(f'Evolucin de {indicator_name} en {country_or_region}')
        plt.xlabel('Ao')
        plt.ylabel(indicator_name)
        plt.show()

# Ejemplo de uso
plot_single_indicator(data_cleaned, 'Access to electricity (% of population)', 'Afghanistan')

# Grfico 2: Comparacin entre la evolucin de varios pases
def plot_multiple_countries(data, indicator_name, countries):
    selected_data = data[(data['Series Name'] == indicator_name) & (data['Country Name'].isin(countries))]
    
    if len(selected_data.columns) > 4 and len(selected_data.iloc[:, 4:].dropna(axis=1, how='all')) > 0:
        plt.figure(figsize=(12, 8))
        for country in countries:
            country_data = selected_data[selected_data['Country Name'] == country]
            plt.plot(country_data.columns[4:], country_data.iloc[0, 4:], label=country)
        
        plt.title(f'Comparacin de {indicator_name} entre pases')
        plt.xlabel('Ao')
        plt.ylabel(indicator_name)
        plt.legend()
        plt.show()

# Ejemplo de uso
plot_multiple_countries(data_cleaned, 'Access to electricity (% of population)', ['Afghanistan', 'Albania', 'Algeria'])
